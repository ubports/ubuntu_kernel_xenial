73 int example_port_default_config(int unit)
74 {
75  opennsl_port_config_t pcfg;
76  opennsl_port_info_t info;
77  int rv;
78  int port;
79  int stp_state = OPENNSL_STG_STP_FORWARD;
80  int stg = 1;
81  int dnx_device = FALSE;
82 
83  dnx_device = example_is_dnx_device(unit);
84  /*
85  * Create VLAN with id DEFAULT_VLAN and
86  * add ethernet ports to the VLAN
87  */
88  opennsl_port_config_t_init(&pcfg);
89 
90  rv = opennsl_port_config_get(unit, &pcfg);
91  if (rv != OPENNSL_E_NONE)
92  {
93  printf("Failed to get port configuration. Error %s\n", opennsl_errmsg(rv));
94  return rv;
95  }
96 
97  /* Set the STP state to forward in default STG for all ports */
98  OPENNSL_PBMP_ITER(pcfg.e, port)
99  {
100  rv = opennsl_stg_stp_set(unit, stg, port, stp_state);
101  if (rv != OPENNSL_E_NONE)
102  {
103  printf("Failed to set STP state for unit %d port %d, Error %s\n",
104  unit, port, opennsl_errmsg(rv));
105  return rv;
106  }
107  }
108 
109  /* Setup default configuration on the ports */
110  opennsl_port_info_t_init(&info);
111 
112  if(dnx_device == FALSE)
113  {
114  info.speed = 0;
115  }
116  info.duplex = OPENNSL_PORT_DUPLEX_FULL;
117  info.pause_rx = OPENNSL_PORT_ABILITY_PAUSE_RX;
118  info.pause_tx = OPENNSL_PORT_ABILITY_PAUSE_TX;
119  info.linkscan = OPENNSL_LINKSCAN_MODE_SW;
120  info.autoneg = FALSE;
121  info.enable = 1;
122 
123  info.action_mask |= ( OPENNSL_PORT_ATTR_AUTONEG_MASK |
124  OPENNSL_PORT_ATTR_DUPLEX_MASK |
125  OPENNSL_PORT_ATTR_PAUSE_RX_MASK |
126  OPENNSL_PORT_ATTR_PAUSE_TX_MASK |
127  OPENNSL_PORT_ATTR_LINKSCAN_MASK |
128  OPENNSL_PORT_ATTR_ENABLE_MASK );
129 
130  if(dnx_device == FALSE)
131  {
132  info.action_mask |= OPENNSL_PORT_ATTR_SPEED_MASK;
133  }
134  OPENNSL_PBMP_ITER(pcfg.e, port)
135  {
136  rv = opennsl_port_selective_set(unit, port, &info);
137  if (OPENNSL_FAILURE(rv))
138  {
139  printf("Failed to set port config for unit %d, port %d, Error %s",
140  unit, port, opennsl_errmsg(rv));
141  return rv;
142  }
143  }
144 
145  return OPENNSL_E_NONE;
146 }

201 char *example_read_user_string(char *buf, size_t buflen)
202 {

204 
205  if (fgets(buf, buflen, stdin) != 0)
206  {
207  size_t len = strlen(buf);

228 int example_read_user_choice(int *choice)
229 {
230  char val;
231  char digits[MAX_DIGITS_IN_CHOICE + 1];
232  int idx = 0;
233  int valid = TRUE;
234 
235  /* parse input string until \n */
236  while((val = getchar()) != '\n')
237  {
238  if ((val >= '0' && val <= '9') && idx < MAX_DIGITS_IN_CHOICE)
239  {
240  digits[idx++] = val;
241  }
242  else
243  {
244  valid = FALSE;
245  }
246  }
247  if ((valid == TRUE) && idx != 0)
248  {
249  digits[idx] = '\0';
250  *choice = atoi(digits);
251  return(OPENNSL_E_NONE);
252  }
253  else
254  {
255  *choice = -1;
256  return(OPENNSL_E_FAIL);
257  }
258 }

269 int opennsl_mac_parse(char *buf, unsigned char *macp)
270 {
271  int i, c1, c2;
272  char *s;
273 #define MAC_ADDR_LEN 17
274 
275  macp[0] = macp[1] = macp[2] = 0;
276  macp[3] = macp[4] = macp[5] = 0;
277 
278  if ((buf == NULL) || (strlen(buf) > MAC_ADDR_LEN)) {
279  return OPENNSL_E_FAIL;
280  }
281 
282  /* skip leading 0x if plain hex format */
283  if (buf[0] == '0' && (buf[1] == 'x' || buf[1] == 'X')) {
284  buf += 2;
285  }
286 
287  /* start at end of string and work backwards */
288  for (s = buf; *s; s++) {
289  ;
290  }
291  for (i = 5; i >= 0 && s >= buf; i--) {
292  c1 = c2 = 0;
293  if (--s >= buf) {
294  if (*s >= '0' && *s <= '9') {
295  c2 = *s - '0';
296  } else if (*s >= 'a' && *s <= 'f') {
297  c2 = *s - 'a' + 10;
298  } else if (*s >= 'A' && *s <= 'F') {
299  c2 = *s - 'A' + 10;
300  } else if (*s == ':') {
301  ;
302  } else {
303  return OPENNSL_E_FAIL;
304  }
305  }
306  if (*s != ':' && --s >= buf) {
307  if (*s >= '0' && *s <= '9') {
308  c1 = *s - '0';
309  } else if (*s >= 'a' && *s <= 'f') {
310  c1 = *s - 'a' + 10;
311  } else if (*s >= 'A' && *s <= 'F') {
312  c1 = *s - 'A' + 10;
313  } else if (*s == ':') {
314  ;
315  } else {
316  return OPENNSL_E_FAIL;
317  }
318  }
319 
320  if (s > buf && s[-1] == ':') {
321  --s;
322  }
323  macp[i] = c1 << 4 | c2;
324  }
325  return OPENNSL_E_NONE;
326 }

358 int opennsl_ip_parse(char *ip_str, unsigned int *ip_val)
359 {
360  unsigned int num = 0, val;
361  char *tok;
362  int count = 0;
363  char buf[16]; /* Maximum length of IP address in dotten notation */
364 
365  if((ip_str == NULL) || (ip_val == NULL) || (strlen(ip_str) > 16))
366  {
367  return -1;
368  }
369  strcpy(buf, ip_str);
370  tok = strtok(buf, ".");
371  while(tok != NULL)
372  {
373  count++;
374  val = atoi(tok);
375  if((val < 0) || (val > 0xff))
376  {
377  return -1;
378  }
379  num = (num << 8) + val;
380  tok = strtok(NULL, ".");
381  }
382  if(count != 4)
383  {
384  return -1;
385  }
386  *ip_val = num;
387 
388  return 0;
389 }

422 int opennsl_ctoi(const char *s, char **end)
423 {
424  unsigned int n, neg;
425  int base = 10;
426 
427  if (s == 0) {
428  if (end != 0) {
429  end = 0;
430  }
431  return 0;
432  }
433 
434  s += (neg = (*s == '-'));
435 
436  if (*s == '0') {
437  s++;
438  if (*s == 'x' || *s == 'X') {
439  base = 16;
440  s++;
441  } else if (*s == 'b' || *s == 'B') {
442  base = 2;
443  s++;
444  } else {
445  base = 8;
446  }
447  }
448 
449  for (n = 0; ((*s >= 'a' && *s < 'a' + base - 10) ||
450  (*s >= 'A' && *s < 'A' + base - 10) ||
451  (*s >= '0' && *s <= '9')); s++) {
452  n = n * base + ((*s <= '9' ? *s : *s + 9) & 15);
453  }
454 
455  if (end != 0) {
456  *end = (char *) s;
457  }
458 
459  return (int) (neg ? -n : n);
460 }

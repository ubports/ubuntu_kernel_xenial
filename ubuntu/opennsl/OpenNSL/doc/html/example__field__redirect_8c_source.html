86 int example_fp_redirect(int unit, int dport)
87 {
88  int rv = OPENNSL_E_NONE;
89  opennsl_field_qset_t qset;
90  opennsl_field_aset_t aset;
91  int pri = OPENNSL_FIELD_GROUP_PRIO_ANY;
92  opennsl_mac_t bpdu_mac_da = MAC_DA;
93  opennsl_mac_t mac_mask = MAC_MASK;
94  int dest_gport;
95  int cos = NEWCOS_VAL;
96 
98  OPENNSL_FIELD_QSET_INIT(qset);
99  OPENNSL_FIELD_QSET_ADD(qset, opennslFieldQualifyDstMac);
100 
102  OPENNSL_FIELD_ASET_INIT(aset);
103  OPENNSL_FIELD_ASET_ADD(aset, opennslFieldActionStat);
104  OPENNSL_FIELD_ASET_ADD(aset, opennslFieldActionRedirectPort);
105  OPENNSL_FIELD_ASET_ADD(aset, opennslFieldActionPrioIntNew);
106 
108  CALL_IF_ERROR_RETURN(opennsl_field_group_create_id(unit, qset, pri, grp));
109 
110  CALL_IF_ERROR_RETURN(opennsl_field_group_action_set(unit, grp, aset));
111 
116  CALL_IF_ERROR_RETURN(opennsl_field_entry_create_id(unit, grp, eid));
117  CALL_IF_ERROR_RETURN(opennsl_field_qualify_DstMac(unit, eid,
118  bpdu_mac_da, mac_mask));
119 
120  CALL_IF_ERROR_RETURN(opennsl_field_stat_create_id(unit, grp, 1,
121  &stats, stat_id));
122 
124  CALL_IF_ERROR_RETURN(opennsl_field_action_add(unit, eid,
125  opennslFieldActionStat, stat_id, 0));
126  OPENNSL_GPORT_LOCAL_SET(dest_gport, dport);
127  CALL_IF_ERROR_RETURN(opennsl_field_action_add(unit, eid,
128  opennslFieldActionRedirectPort, 0, dest_gport));
129  CALL_IF_ERROR_RETURN(opennsl_field_action_add(unit, eid,
130  opennslFieldActionPrioIntNew, cos, 0));
131 
132  rv = opennsl_field_group_install(unit, grp);
133  if (OPENNSL_E_NONE != rv)
134  {
135  printf("Failed to install the group, Error %s\n", opennsl_errmsg(rv));
136  CALL_IF_ERROR_RETURN(opennsl_field_group_destroy(unit, grp));
137  return rv;
138  }
139 
140  printf("Packet qualification criteria: \n");
141  l2_print_mac(" ==> MAC DA -", bpdu_mac_da);
142  printf("\n");
143  printf("Configured system to count and redirect qualified traffic to "
144  "port %d", dport);
145 
146  return rv;
147 }

158 int main(int argc, char *argv[])
159 {
160  int rv = OPENNSL_E_NONE;
161  int choice;
162  int unit = DEFAULT_UNIT;
163  uint64 val;
164  int port;
165 
166  if((argc != 1) || ((argc > 1) && (strcmp(argv[1], "--help") == 0))) {
167  printf("%s\n\r", example_usage);
168  return OPENNSL_E_PARAM;
169  }
170 
171  /* Initialize the system */
172  rv = opennsl_driver_init((opennsl_init_t *) NULL);
173 
174  if(rv != OPENNSL_E_NONE) {
175  printf("\r\nFailed to initialize the system.\r\n");
176  return rv;
177  }
178 
179  rv = example_port_default_config(unit);
180  if (rv != OPENNSL_E_NONE) {
181  printf("\r\nFailed to apply default config on ports, rc = %d (%s).\r\n",
182  rv, opennsl_errmsg(rv));
183  }
184 
185  /* Add ports to default vlan. */
186  printf("Adding ports to default vlan.\r\n");
187  rv = example_switch_default_vlan_config(unit);
188  if(rv != OPENNSL_E_NONE) {
189  printf("\r\nFailed to add ports to default VLAN. Error: %s\r\n",
190  opennsl_errmsg(rv));
191  }
192 
193  while(1) {
194  printf("\r\nUser menu: Select one of the following options\r\n");
195  printf("1. Apply filter to redirect traffic\n");
196  printf("2. Remove redirection filter\n");
197  printf("3. Display the number of packets redirected\n");
198 #ifdef INCLUDE_DIAG_SHELL
199  printf("9. Launch diagnostic shell\n");
200 #endif
201  printf("0. Quit the application.\r\n");
202 
203  if(example_read_user_choice(&choice) != OPENNSL_E_NONE)
204  {
205  printf("Invalid option entered. Please re-enter.\n");
206  continue;
207  }
208  switch(choice)
209  {
210  case 1:
211  {
212  printf("Enter port number to which traffic needs to be redirected:\n");
213  if(example_read_user_choice(&port) != OPENNSL_E_NONE)
214  {
215  printf("Invalid option entered. Please re-enter.\n");
216  continue;
217  }
218  rv = example_fp_redirect(unit, port);
219  if (rv != OPENNSL_E_NONE) {
220  printf("Failed to configure the system for traffic redirection, "
221  "Error %s\n", opennsl_errmsg(rv));
222  return rv;
223  }
224  break;
225  }
226 
227  case 2:
228  {
229  CALL_IF_ERROR_RETURN(opennsl_field_entry_destroy(unit, eid));
230  CALL_IF_ERROR_RETURN(opennsl_field_group_destroy(unit, grp));
231  printf("Removed the redirection filter\n");
232  break;
233  }
234 
235  case 3:
236  {
237  rv = opennsl_field_stat_get(unit, stat_id, stats, &val);
238  if (rv != OPENNSL_E_NONE) {
239  printf("Failed to get the statistics, "
240  "Error %s\n", opennsl_errmsg(rv));
241  return rv;
242  }
243  printf("Number of packets redirected: %lld (0x%0x)\n", val, val);
244  break;
245  }
246 
247 #ifdef INCLUDE_DIAG_SHELL
248  case 9:
249  {
250  opennsl_driver_shell();
251  break;
252  }
253 #endif
254 
255  case 0:
256  {
257  printf("Exiting the application.\n");
258  return OPENNSL_E_NONE;
259  }
260  default:
261  break;
262  } /* End of switch */
263  } /* End of while */
264 
265  return rv;
266 }

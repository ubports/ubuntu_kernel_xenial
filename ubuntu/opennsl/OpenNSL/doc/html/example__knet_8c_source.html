96 void example_knet_database_update(int port, char *ifName, int nID, int fID)
97 {
98  int i;
99  int entryFound = 0;
100 
101  for(i = 0; i < KNET_INTF_COUNT; i++)
102  {
103  /* Check if the entry for the given port is already present */
104  if(port == knet_if_data[i].port)
105  {
106  entryFound = 1;
107  break;
108  }
109  }
110 
111  if(entryFound == 0)
112  {
113  /* Find the first free entry */
114  for(i = 0; i < KNET_INTF_COUNT; i++)
115  {
116  if(knet_if_data[i].port == 0)
117  {
118  break;
119  }
120  }
121  }
122 
123  knet_if_data[i].port = port;
124  knet_if_data[i].netIfID = nID;
125  knet_if_data[i].filterID = fID;
126  strcpy(knet_if_data[i].name, ifName);
127 
128  /* Updated the new entry. So, increment KNET interface count */
129  if(entryFound == 0)
130  {
131  knet_intf_count++;
132  }
133  return;
134 }

166 /******************************************************************************
167  * \brief Creates linux interfaces and corresponding KNET filters to map
168  * each front panel interface to these linux interfaces.
169  *
170  * \param port [IN] port number
171  * \param unit [IN] unit number
172  * \param ifName [IN] interface name
173  *
174  * \return None
175  *
176  *****************************************************************************/
177 int example_linux_interface_create(int unit, int port, char *ifName)
178 {
179  int rv;
180  opennsl_knet_netif_t netif;
181  opennsl_knet_filter_t filter;
182  unsigned char baseMac[6] = { 0x02, 0x10, 0x18, 0x00, 0x00, 0x01 };
183 
184  opennsl_knet_netif_t_init(&netif);
185  netif.type = OPENNSL_KNET_NETIF_T_TX_LOCAL_PORT;
186  netif.port = port; /* Egress port */
187  strcpy(netif.name, ifName); /* Copy the base interface name */
188  /* Set MAC address associated with this interface */
189  memcpy(netif.mac_addr, baseMac, 6);
190  rv = opennsl_knet_netif_create(unit, &netif);
191  if (rv >= 0)
192  {
193  printf("Created Interface ID %d for interface name \"%s\" \n",
194  netif.id, ifName);
195  }
196  else
197  {
198  printf("rv = %d Error creating KNET interface \"%s\" \n", rv, ifName);
199  return rv;
200  }
201 
202  /* Add filter for packets from port */
203  opennsl_knet_filter_t_init(&filter);
204 
205  filter.type = OPENNSL_KNET_FILTER_T_RX_PKT;
206 
207  /*
208  This filter is modified to **NOT** STRIP VLAN TAG when this port is
209  attached to a bridge
210  */
211 
212  filter.flags = OPENNSL_KNET_FILTER_F_STRIP_TAG;
213  filter.priority = 0;
214  /* Send packet to network interface */
215  filter.dest_type = OPENNSL_KNET_DEST_T_NETIF;
216  filter.dest_id = netif.id;
217  filter.match_flags = OPENNSL_KNET_FILTER_M_INGPORT;
218  filter.m_ingport = port;
219 
220  rv = opennsl_knet_filter_create(unit, &filter);
221  if (rv >= 0)
222  {
223  printf("Created Filter ID %d for interface name \"%s\" \n",
224  filter.id, ifName);
225  }
226  else
227  {
228  printf("rv = %d Error creating filter for interface \"%s\" \n",
229  rv, ifName);
230  return rv;
231  }
232 
233  /* Update the database */
234  example_knet_database_update(port, ifName, netif.id, filter.id);
235 
236  return rv;
237 }

240 /******************************************************************************
241  * \brief Configure IP address and netmask on the KNET interface.
242  *
243  * \param port [IN] port number
244  * \param unit [IN] unit number
245  * \param ifName [IN] interface name
246  *
247  * \return None
248  *
249  *****************************************************************************/
250 void example_ip_config(char *ifName, char* ip, char* netmask)
251 {
252  char cmd[100];
253 
254  memset(cmd, 0, sizeof(cmd));
255 
256  sprintf(cmd, "ifconfig %s %s netmask %s", ifName, ip, netmask);
257  if(system(cmd) == -1)
258  {
259  printf("Failed to configure interface \"%s\" with IP: %s Netmask: %s\n",
260  ifName, ip, netmask);
261  return;
262  };
263 
264  sprintf(cmd, "ifconfig %s up > /dev/null", ifName);
265  if(system(cmd) == -1)
266  {
267  printf("Failed to bringup interface \"%s\"\n", ifName);
268  return;
269  };
270 
271  printf("Configured interface \"%s\" with IP: %s Netmask: %s\n",
272  ifName, ip, netmask);
273 }

284 void example_knet_intf_create(int unit)
285 {
286  int port;
287  int rv;
288  char ifName[OPENNSL_KNET_NETIF_NAME_MAX];
289  char ip[IP_ADDR_LEN];
290  char netmask[NETMASK_LEN];
291 
292  if(knet_intf_count == KNET_INTF_COUNT)
293  {
294  printf("Maximum number of KNET interfaces are already present.\r\n");
295  return;
296  }
297 
298  printf("\n\rEnter port number: \n");
299  if(example_read_user_choice(&port) != OPENNSL_E_NONE)
300  {
301  printf("Invalid option entered.\n");
302  return;
303  }
304  printf("\n\rEnter knet interface name: ");
305  if(example_read_user_string(ifName, sizeof(ifName)) == NULL)
306  {
307  printf("\n\rInvalid KNET interface is entered.\n\r");
308  return;
309  }
310  printf("\n\rEnter IP address in dotted decimal format: ");
311  if(example_read_user_string(ip, sizeof(ip)) == NULL)
312  {
313  printf("\n\rInvalid IP address is entered.\n\r");
314  return;
315  }
316  printf("\n\rEnter IP netmask in dotted decimal format: ");
317  if(example_read_user_string(netmask, sizeof(netmask)) == NULL)
318  {
319  printf("\n\rInvalid IP netmask is entered.\n\r");
320  return;
321  }
322 
323  /* Create Linux interface. */
324  rv = example_linux_interface_create(unit, port, ifName);
325  if(rv < 0) {
326  printf("Failed to create Linux interface \"%s\". rv: %d\r\n", ifName, rv);
327  return;
328  }
329 
330  printf("Created Linux interface \"%s\" successfully.\r\n", ifName);
331  example_ip_config(ifName, ip, netmask);
332  return;
333 }

336 /******************************************************************************
337  * \brief To delete KNET interface.
338  *
339  * \param unit [IN] unit number
340  *
341  * \return None
342  *
343  *****************************************************************************/
344 void example_knet_intf_delete(int unit)
345 {
346  int rv;
347  int i;
348  char ifName[OPENNSL_KNET_NETIF_NAME_MAX];
349 
350  if(knet_intf_count == 0)
351  {
352  printf("\n\rThere are no KNET interfaces present in the system.\n\r");
353  }
354  printf("\n\rEnter KNET interface name: ");
355  if(example_read_user_string(ifName, sizeof(ifName)) == NULL)
356  {
357  printf("\n\rInvalid KNET interface is entered.\n\r");
358  return;
359  }
360 
361  for(i = 0; i < KNET_INTF_COUNT; i++)
362  {
363  if(strcmp(ifName, knet_if_data[i].name) == 0)
364  {
365  break;
366  }
367  }
368 
369  if (i == KNET_INTF_COUNT)
370  {
371  printf("\n\rThe specific interface is not found.\n\r");
372  return;
373  }
374 
375  /* Destroy KNET interface. */
376  rv = opennsl_knet_netif_destroy(unit, knet_if_data[i].netIfID);
377  if(rv < 0) {
378  printf("Error destroying network interface: %s\n",
379  opennsl_errmsg(rv));
380  return;
381  }
382 
383  rv = opennsl_knet_filter_destroy(unit, knet_if_data[i].filterID);
384  if(rv < 0) {
385  printf("Error destroying packet filter: %s\n",
386  opennsl_errmsg(rv));
387  return;
388  }
389 
390  /* Clear the database for the deleted port */
391  memset(&knet_if_data[i], 0, sizeof(knet_if_data_t));
392  knet_intf_count--;
393 }

403 int main(int argc, char *argv[])
404 {
405  int rv = 0;
406  int unit = DEFAULT_UNIT;
407  int choice;
408 
409  if((argc != 1) || ((argc > 1) && (strcmp(argv[1], "--help") == 0))) {
410  printf("%s\n\r", example_usage);
411  return OPENNSL_E_PARAM;
412  }
413 
414  /* Initialize the system */
415  rv = opennsl_driver_init((opennsl_init_t *) NULL);
416 
417  if(rv != OPENNSL_E_NONE) {
418  printf("\r\nFailed to initialize the system. Error %s\r\n",
419  opennsl_errmsg(rv));
420  return rv;
421  }
422 
423  /* cold boot initialization commands */
424  rv = example_port_default_config(unit);
425  if (rv != OPENNSL_E_NONE) {
426  printf("\r\nFailed to apply default config on ports, rc = %d (%s).\r\n",
427  rv, opennsl_errmsg(rv));
428  }
429 
430  /* Add ports to default vlan. */
431  printf("Adding ports to default VLAN %d.\r\n", DEFAULT_VLAN);
432  rv = example_switch_default_vlan_config(unit);
433  if(rv != OPENNSL_E_NONE) {
434  printf("\r\nFailed to add ports to default VLAN %d. rv: %d\r\n",
435  DEFAULT_VLAN, rv);
436  return rv;
437  }
438 
439  while(1) {
440 
441  printf("\nUser Menu: Select one of the following options\n");
442  printf("1. Create KNET interface\n");
443  printf("2. Delete KNET interface\n");
444  printf("3. Display KNET interface database\n");
445 #ifdef INCLUDE_DIAG_SHELL
446  printf("9. Launch diagnostic shell\n");
447 #endif
448  printf("0. Quit the application.\n");
449 
450  if(example_read_user_choice(&choice) != OPENNSL_E_NONE)
451  {
452  printf("Invalid option entered. Please re-enter.\n");
453  continue;
454  }
455  switch(choice) {
456  case 1:
457  {
458  example_knet_intf_create(unit);
459  break;
460  }
461 
462  case 2:
463  {
464  example_knet_intf_delete(unit);
465  break;
466  }
467 
468  case 3:
469  {
470  example_knet_database_show();
471  break;
472  }
473 
474 #ifdef INCLUDE_DIAG_SHELL
475  case 9:
476  {
477  opennsl_driver_shell();
478  break;
479  }
480 #endif
481 
482  case 0:
483  {
484  printf("Exiting the application.\n");
485  return OPENNSL_E_NONE;
486  }
487 
488  default:
489  break;;
490  } /* End of switch */
491  } /* End of while */
492 
493  return rv;
494 }

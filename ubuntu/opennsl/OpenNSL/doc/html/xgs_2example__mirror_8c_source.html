173 static int example_mirror_setup(int unit, int mode, int srcPort, int probePort)
174 {
175  int rv;
176  opennsl_gport_t gport;
177  unsigned int flags = 0;
178  opennsl_mirror_destination_t mirror_dest;
179 
180  rv = opennsl_mirror_init(unit);
181  if(rv != OPENNSL_E_NONE) {
182  printf("\r\nFailed to init mirror module. Error: %s\r\n",
183  opennsl_errmsg(rv));
184  }
185 
186  opennsl_mirror_destination_t_init(&mirror_dest);
187  OPENNSL_GPORT_MODPORT_SET(gport, 0, probePort);
188  mirror_dest.gport = gport;
189 
190  rv = opennsl_mirror_destination_create(unit, &mirror_dest);
191  if(rv != OPENNSL_E_NONE) {
192  printf("\r\nFailed to create mirror dest. Error: %s\r\n",
193  opennsl_errmsg(rv));
194  }
195 
196  /* Configure the source port and mirror to port along with mirror mode */
197  switch (mode)
198  {
199  case 1:
200  /* Ingress mirroring only */
201  flags |= OPENNSL_MIRROR_PORT_INGRESS;
202  break;
203 
204  case 2:
205  /* Egress mirroring only */
206  flags |= OPENNSL_MIRROR_PORT_EGRESS;
207  break;
208 
209  case 3:
210  /* Both Ingress and Egress mirroring */
211  flags |= (OPENNSL_MIRROR_PORT_INGRESS | OPENNSL_MIRROR_PORT_EGRESS);
212  break;
213 
214  default:
215  break;
216  }
217 
218  rv = opennsl_mirror_port_dest_add(unit, srcPort, flags, mirror_dest.mirror_dest_id);
219  if(rv != OPENNSL_E_NONE) {
220  printf("\r\nFailed to enable mirroring on the port. Error: %s\r\n",
221  opennsl_errmsg(rv));
222  }
223 
224  return rv;
225 }
